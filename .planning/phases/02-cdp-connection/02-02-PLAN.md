The main flow (same as agent-browser):
```
fn main() {
    let cli = Cli::parse();
    // ensure daemon is running
    connection::ensure_daemon(&cli.session)?;
    // build command JSON
    let cmd = match &cli.command { ... };
    // send to daemon
    let response = connection::send_command(cmd, &cli.session)?;
    // format and print response
    output::print_response(&response, &cli.format);
}
```
